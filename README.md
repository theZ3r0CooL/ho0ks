## Install

```bash
npm install --save ho0ks
```

## Usage

```tsx
import * as React from 'react'

import { useMyHook } from 'ho0ks'

const Example = () => {
  const example = useMyHook()
  return (
    <div>
      {example}
    </div>
  )
}
```
